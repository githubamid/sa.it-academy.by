[update_centos]
192.168.100.7 ansible_user=jump_user

ssh-copy-id -i ~/.ssh/id_rsa.pub jump_user@192.168.100.7

# Config file for remote_servers

```bash
Host jump_server
      User jump_user
      HostName 192.168.100.7
Host 10.0.2.*
      ProxyJump jump_server
```
